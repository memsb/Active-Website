// put full path to Smarty.class.php
require('/usr/local/lib/php/Smarty/Smarty.class.php');

class test {

	private $logged = False;

	public function get(){
	}

	public function post(){
		$this->logged = True;
	}

	public function visit(){
		$smarty = new Smarty();

		$smarty->setTemplateDir('/var/www/Active-Website/smarty/templates');
		$smarty->setCompileDir('/var/www/Active-Website/smarty/templates_c');
		$smarty->setCacheDir('/var/www/Active-Website/smarty/cache');
		$smarty->setConfigDir('/var/www/Active-Website/smarty/configs');


		$page = $smarty->createTemplate('index.tpl');
		$page->assign('title', 'Page');

		if( $this->logged ){
			$right = $smarty->createTemplate('menu.tpl');
		} else{
			$right = $smarty->createTemplate('account.tpl');
		}
		$page->assign('right', $right->fetch());

		$content = $smarty->createTemplate('post.tpl');		
		$content->assign('title', 'Title');
		$content->assign('name', 'Name');
		$content->assign('date', '25th January 2012');
		$content->assign('body', 'WEEEEEEEEEEeeeeeeeeeeeeeeeeee');
		$page->assign('content', $content->fetch());
		$page->display();
	}
}

$test = new test();

if( count($_POST) != 0 )
	$test->post();

if( count($_GET) != 0 )
	$test->get();

$test->visit();

?>
